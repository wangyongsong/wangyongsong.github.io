# 自己封装的hook

# 1、导出列表 - 后端导出

# 场景

ERP系统，列表的导出功能，内置函数节流，防止误操作

本来做想做导出条数的限制，后来想想估计用不到，所以注释了

# 封装文件(useExport.tsx)

import { useState } from 'react';
import { message, Modal } from 'antd';
import lodash from 'lodash';
import { ExclamationCircleOutlined } from '@ant-design/icons';
// import { useAccess } from 'umi';

const { confirm } = Modal;

const useExport = (formData: any, fetchFn: any) => {
  const [searchFormData, setsearchFormData] = useState(formData);
  // const access: any = useAccess();

  const setformData = (v: any) => {
    setsearchFormData({ ...v });
  };

  const confirmFn = lodash.throttle(
    async () => {
      const copyData = lodash.cloneDeep(searchFormData);
      delete copyData.current;
      delete copyData.pageSize;
      delete copyData.total;
      const res = await fetchFn({ ...copyData });
      if (res.success) message.success(`${res.message} 请在下载中心查看进度！`);
    },
    3000,
    { leading: true, trailing: false },
  );

  const exportFn = () => {
    const total = searchFormData.total || 0;
    confirm({
      title: '确认导出这些数据吗?',
      icon: <ExclamationCircleOutlined />,
      content: `共 ${total} 条数据！`,
      onOk() {
        // const limitNum = access[accessName] || 0
        // const ok = total < limitNum || total === limitNum
        if (total === 0) {
          message.warning('无导出数据，导出条数为0');
          return;
        }
        confirmFn();
        // if(ok) {
        //     confirmFn()
        // } else {
        //     message.error(`导出权限条数：${limitNum}条，已超出限制，请重试`)
        // }
      },
    });
  };

  return [setformData, exportFn, searchFormData];
};

export default useExport;
